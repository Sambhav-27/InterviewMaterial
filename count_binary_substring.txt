#include <iostream>
using namespace std;

// nC2= n*(n-1)/2;

int main() {
	int t,i,j,k,n,m;
	cin>>t;
	while(t--){
	    int ans=0;
	    string s;
	    cin>>s;
	    for(i=0;i<s.length();++i)
	    if(s[i]=='1') ans++;
	    
	    cout<<ans*(ans-1)/2<<"\n";
	}
	
	return 0;
}

Given a binary string, count number of substrings that start and end with 1. For example, if the input string is “00100101”, then there are three substrings “1001”, “100101” and “101”.
